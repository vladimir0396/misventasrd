<script>

const contraseñaAdmin = "1234"; // 🔥 Cambia esto por la que tú quieras

function verificarClave() {
    let clave = document.getElementById("clave").value;

    if (clave === contraseñaAdmin) {
        document.getElementById("login").style.display = "none";
        document.getElementById("panel").style.display = "block";
    } else {
        alert("Contraseña incorrecta");
    }
}

function guardarEnvio() {
    let nuevo = document.getElementById("nuevoEnvio").value;
    localStorage.setItem("envioGeneral", nuevo);
    alert("Costo de envío actualizado");
}

function guardarPrecios() {
    let camisa = document.getElementById("precioCamisa").value;
    let zapatos = document.getElementById("precioZapatos").value;

    localStorage.setItem("precioCamisa", camisa);
    localStorage.setItem("precioZapatos", zapatos);

    alert("Precios actualizados");
}

</script>
